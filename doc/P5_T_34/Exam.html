def to_s
  aux = "Titulo del examen: " + @title + "\n"
  aux += "------------------------------------------------- \n"
  act = @top
  cont = 0
  
  while (act != nil)
           aux += cont.to_s + ") " + act.value.to_s
           aux += " \n ++++++++++++++++++++++++++++++++++ \n"
           act = act.next
           cont += 1
  end
  return aux
end
